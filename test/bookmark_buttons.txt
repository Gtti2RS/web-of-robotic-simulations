- This file provides two scripts that help when observing properties in web browser.
- Create 2 bookmarks and paste following scripts in the URL field and set bookmarks toolbar always visible.
- When observing the properties, you can click the bookmarks as buttons to scroll the contents 
automatically like in terminal and clear the contents

#tail
javascript:(()=>{if(window.__TAIL_STOP){window.__TAIL_STOP();alert('Tail OFF');return}const S={};const scrollable=e=>e&&e.scrollHeight>e.clientHeight;const parentScrollable=n=>{let e=n;while(e&&e!==document.body){if(scrollable(e))return e;e=e.parentElement}return document.scrollingElement||document.documentElement};const atBottom=e=>e.scrollHeight-e.clientHeight-e.scrollTop<4;const toBottom=e=>{e.scrollTop=e.scrollHeight};S.el=document.scrollingElement||document.documentElement;S.follow=true;S.onScroll=()=>{S.follow=atBottom(S.el)};S.el.addEventListener('scroll',S.onScroll,false);S.ob=new MutationObserver(()=>{if(S.follow)toBottom(S.el)});S.ob.observe(document.body,{childList:true,subtree:true,characterData:true});S.onKey=e=>{if((e.ctrlKey||e.altKey)&&e.key.toLowerCase()==='p'){S.follow=!S.follow;console.log('Tail follow:',S.follow)}};window.addEventListener('keydown',S.onKey,false);S.onPick=e=>{if(e.altKey){S.el.removeEventListener('scroll',S.onScroll,false);S.el=parentScrollable(e.target);S.el.addEventListener('scroll',S.onScroll,false);S.follow=atBottom(S.el);console.log('Tail target set:',S.el);e.preventDefault()}};window.addEventListener('click',S.onPick,true);toBottom(S.el);window.__TAIL_STOP=()=>{S.ob.disconnect();window.removeEventListener('keydown',S.onKey,false);window.removeEventListener('click',S.onPick,true);S.el.removeEventListener('scroll',S.onScroll,false);delete window.__TAIL_STOP;};console.log('Tail ON â€” Alt+Click a pane to follow. Ctrl/Alt+P toggles follow. Run bookmarklet again to stop.');})()
